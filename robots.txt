# 语法规则：
#   User-agent: 指定适用的搜索引擎爬虫名称
#   Allow: 允许抓取的路径
#   Disallow: 禁止抓取的路径
#   Sitemap: XML 网站地图地址
# 
# 常见爬虫名称：
#   Googlebot - 谷歌爬虫
#   Baiduspider - 百度爬虫
#   Bingbot - 必应爬虫
#   360Spider - 360 搜索爬虫
#   SougouSpider - 搜狗爬虫
#   * - 所有爬虫
# =========================================================================

# 允许所有搜索引擎爬虫访问
User-agent: *

# 允许抓取 HTML 页面
Allow: /*.html$

# 允许抓取 CSS 样式表（影响页面渲染质量评估）
Allow: /css/*.css

# 允许抓取 JavaScript 文件（影响交互式内容理解）
Allow: /js/*.js

# 允许抓取图片资源（用于图片搜索）
Allow: /images/*.jpg
Allow: /images/*.jpeg
Allow: /images/*.png
Allow: /images/*.gif
Allow: /images/*.webp
Allow: /images/*.svg

# 禁止抓取敏感文件和后台路径
Disallow: /admin/
Disallow: /backend/
Disallow: /config/
Disallow: /private/
Disallow: /tmp/
Disallow: /*.log$
Disallow: /*.sql$
Disallow: /*.bak$
Disallow: /*.zip$
Disallow: /*.rar$

# 禁止抓取 URL 参数（防止重复内容，除非必要）
